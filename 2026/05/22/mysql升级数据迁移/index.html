mysql升级数据迁移

mysql升级数据迁移

一、需求

客户升级采用并行方案，需要将老版本数据导入新版本中，此为数据迁移方案

二、思路

1、 组织数据可以采用先同步源库数据到目标库，然后再运行产品升级sql

2、业务数据，可以直接用sql带字段插入

三、解决方案

方案：采用生成数据sql的模式。理清需要同步的表。

四、实现步骤

-- 设置参数

SET @oldDB = 'oldDB';

SET @newDB = 'newDB';

-- 生成业务数据迁移SQL，以应用前缀匹配

set @formName ="ixizx_";

SELECT CONCAT(

'INSERT INTO ', @newDB, '.', TABLE_NAME, ' (',

GROUP_CONCAT(COLUMN_NAME ORDER BY ORDINAL_POSITION),

') ',

'\nSELECT ',

GROUP_CONCAT(COLUMN_NAME ORDER BY ORDINAL_POSITION),

' FROM ', @oldDB, '.', TABLE_NAME, ';'

) AS generated_sql

FROM information_schema.COLUMNS

WHERE TABLE_SCHEMA = @oldDB

AND TABLE_NAME LIKE CONCAT(@formName, '%')

GROUP BY TABLE_NAME;

-- 系统表，精确到表名

SET @tableNames = 'h_org_user,h_org_dept_user,h_org_department,h_org_role,h_org_role_group,h_org_role_user,h_org_user_extend_attr,h_org_user_union_extend_attr,h_related_corp_setting';

-- 生成迁移SQL

SELECT

CONCAT(

'DROP TABLE IF EXISTS ', @newDB, '.', TABLE_NAME, '; ',

'CREATE TABLE ', @newDB, '.', TABLE_NAME,

' LIKE ', @oldDB, '.', TABLE_NAME, '; ',

'INSERT INTO ', @newDB, '.', TABLE_NAME,

' SELECT * FROM ', @oldDB, '.', TABLE_NAME, ';'

) AS migration_sql

FROM INFORMATION_SCHEMA.TABLES

WHERE TABLE_SCHEMA = @oldDB

AND FIND_IN_SET(TABLE_NAME, @tableNames) > 0;
