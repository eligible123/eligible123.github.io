低代码rpa融合

低代码rpa融合

一、需求

低代码融合rpa，利用rpa配置自动化数据抓取，然后将数据传到低代码，用于展示,汇总,审核。可能针对一个模型配置多个抓取动作，需要支持多次数据传递，特别是子表。

二、思路

1、两边需要先同步人员，采用手机号唯一

2、先在低代码建立模型，rpa那边抓取模型配置，然后在rpa配置自动化抓取数据和低代码模型对应关系，抓取完成回传至低代码。

3、确定唯一键id，方便多次传值。

三、解决方案

方案：采用openapi授权模式，提供组织及用户接口，同步用户。提供模型及字段获取接口，给rpa能获取到模型配置。提供数据写入接口。

四、实现步骤

1、后台增加openapi配置，rpa能获取到token

后台管理-api接口配置 openapi模式，通过接口获取token

http://xxxx/api/oauth/token

请求方式 post 参数 grant_type,client_id,client_secret

package com.authine.cloudpivot.web.api.controller.openapi.v2;

import com.authine.cloudpivot.engine.api.model.bizmodel.BizPropertyModel; import com.authine.cloudpivot.engine.api.model.bizmodel.BizSchemaModel; import com.authine.cloudpivot.engine.api.model.organization.UserModel; import com.authine.cloudpivot.engine.api.model.runtime.BizObjectCreatedModel; import com.authine.cloudpivot.engine.api.model.runtime.BizObjectModel; import com.authine.cloudpivot.engine.domain.bizmodel.BizSchema; import com.authine.cloudpivot.engine.enums.type.BizPropertyType; import com.authine.cloudpivot.engine.repository.bizmodel.BizSchemaExtRepository; import com.authine.cloudpivot.ext.domain.RpaWriteBack; import com.authine.cloudpivot.web.api.controller.base.BaseController; import com.authine.cloudpivot.web.api.handler.CustomizedOrigin; import com.authine.cloudpivot.web.api.view.ResponseResult; import io.swagger.annotations.Api; import io.swagger.annotations.ApiOperation; import lombok.extern.slf4j.Slf4j; import org.apache.commons.lang3.StringUtils; import org.json.JSONObject; import org.springframework.beans.factory.annotation.Autowired; import org.springframework.web.bind.annotation.*;

import java.util.ArrayList; import java.util.HashMap; import java.util.List; /** @author *zhengyj @date *2022/6/23 @description *rpa */@Api( tags = {"rpa::rpa"} )@RestController @RequestMapping({"/openapi/v2/rpa"})@CustomizedOriginpublic class RPAController extends BaseController {

String adminId = "2c9280a26706a73a016706a93ccf002b";

@Autowired private BizSchemaExtRepository bizSchemaExtRepository;

@ApiOperation("获取模型字段") @RequestMapping(value = "/getModelFileds", method = RequestMethod.GET) public ResponseResult getModelFileds(@RequestParam String id){

List> filedList = new ArrayList<>(); try { BizSchema bizSchema = bizSchemaExtRepository.getById(id); if (null != bizSchema){ id = bizSchema.getCode(); } } catch (Exception e){ }

BizSchemaModel listBySchemaCode = getBizSchemaFacade().getBySchemaCodeWithProperty(id,true); for (BizPropertyModel bizPropertyModel : listBySchemaCode.getProperties()) { // 去除系统字段 if (!bizPropertyModel.getDefaultProperty()){ HashMap filedMap = new HashMap<>(); filedMap.put("code",bizPropertyModel.getCode()); filedMap.put("name",bizPropertyModel.getName()); filedMap.put("type",bizPropertyModel.getPropertyType()); // 子表封装 if (BizPropertyType.CHILD_TABLE.equals(bizPropertyModel.getPropertyType())){ List> childPropertys = new ArrayList<>(); for (BizPropertyModel childBizPropertyModel : bizPropertyModel.getSubSchema().getProperties()) { if (!childBizPropertyModel.getDefaultProperty()){ HashMap childProperty = new HashMap<>(); childProperty.put("code",childBizPropertyModel.getCode()); childProperty.put("name",childBizPropertyModel.getName()); childProperty.put("type",childBizPropertyModel.getPropertyType()); childPropertys.add(childProperty); }

} filedMap.put("child",childPropertys); } filedList.add(filedMap); } } return getOkResponseResult(filedList,"获取模型字段成功"); }

@ApiOperation("回写") @RequestMapping(value = "/dataWriteBack", method = RequestMethod.POST) public ResponseResult dataWriteBack(@RequestBody RpaWriteBack rpaWriteBack){ String id = rpaWriteBack.getId(); String userId = rpaWriteBack.getUserId(); HashMap data = rpaWriteBack.getData(); if (StringUtils.isEmpty(id)|| StringUtils.isEmpty(userId)){ return getErrResponseResult(600L,"id或者user为空"); } String bizObjectId = (String) data.get("id");

if (StringUtils.isEmpty(bizObjectId)){ return getErrResponseResult(600L,"data.id为空"); } else { try { BizSchema bizSchema = bizSchemaExtRepository.getById(id); if (null != bizSchema){ id = bizSchema.getCode(); } } catch (Exception e){} // 增加锁，避免重复调用 synchronized (bizObjectId){ BizObjectCreatedModel bizObject = getBizObjectFacade().getBizObject(adminId, id, bizObjectId);

// update if (null!=bizObject&&bizObject.getData().size()>2){ for (String filed : data.keySet()) { bizObject.getData().put(filed,data.get(filed)); } getBizObjectFacade().saveBizObject(userId, bizObject, false);

} else { data.put("sequenceStatus","COMPLETED"); BizObjectModel bizObjectModel = new BizObjectModel(id,data,false); getBizObjectFacade().saveBizObject(userId, bizObjectModel, false); } } } return getOkResponseResult(bizObjectId,"数据回写成功"); }

}
