easyexcel 实现导入+数据校验

一.背景

导入数据时，需要针对数据做一些定开的数据校验，并对后续云枢的升级更为友好型。

二.思路/方案

1.采用 easyexcel 作为基础框架，实现高性能 可扩展的 数据导入。

  • 使用监听器 来实现多个数据模版的业务实现：

    • Controller → BasicExcelFactory → AbstractBasicExcelService → AbstractBasicImportListenerService → 具体业务实现

    三.实现过程。

    - 核心组件 BasicExcelFactory：工厂类，根据类型获取具体服务

    AbstractBasicExcelService：抽象服务，提供模板方法

    AbstractBasicImportListenerService：基础监听器，处理通用逻辑

    具体业务实现类：如KlCangkuBasicExcelService等

    1.1 BasicExcelFactory 使用工厂模式根据类型动态获取对应的服务实现：

    @Component

    @Slf4j

    public class BasicExcelFactory {

    private AbstractBasicExcelService getExcelService(String type) {

    AbstractBasicExcelService service = null;

    switch (Objects.requireNonNull(BasicExcelEnum.getByType(type))) {

    case KL_XCWZ:

    service = SpringContextUtils.getBean("klxcwzBasicExcelService");

    break;

    case KL_CANGKU:

    service = SpringContextUtils.getBean("klCangkuBasicExcelService");

    break;

    case KL_ZZP:

    service = SpringContextUtils.getBean("klZzpBasicExcelService");

    break;

    default:

    break;

    }

    return service;

    }

    public ImportResult doImport(UploadInitModel initModel, MultipartFile file) throws IOException {

    AbstractBasicExcelService service = getExcelService(initModel.getType());

    if (service == null) {

    String errorMsg = "数据导入失败，类型匹配异常！";

    log.error(errorMsg);

    return new ImportResult(initModel.getTaskId(), errorMsg, ImportResult.allFailFlag);

    }

    return service.doImportTask(file, initModel);

    }

    public void doExportErrorData(String type, String taskId, HttpServletResponse response) {

    AbstractBasicExcelService service = getExcelService(type);

    if (service == null) {

    String errorMsg = "基础档案导入，类型匹配异常！";

    log.error(errorMsg);

    return;

    }

    service.exportErrorData(type, taskId, response);

    }

    }

    1.2 AbstractBasicImportListenerService 继承自EasyExcel的AnalysisEventListener，实现了核心导入逻辑：

    public abstract class AbstractBasicImportListenerService extends AnalysisEventListener {

    protected RedissonClient redissonClient; protected BasicExcelHelper helper; protected NamedParameterJdbcTemplate namedParameterJdbcTemplate; protected MaterialCostListService materialCostListService; // protected UserService userService;

    public AbstractBasicImportListenerService(BasicInfoContext context) { this.redissonClient = context.getRedissonClient(); this.helper = context.getHelper(); this.namedParameterJdbcTemplate=context.getNamedParameterJdbcTemplate(); this.materialCostListService = context.getMaterialCostListService(); // this.userService = context.getUserService(); }

    protected abstract String getType();

    protected abstract CommonImportValidateRes doValidateAndInit();

    protected abstract List> buildDataMap(List successData);

    protected abstract String getErrorRedisKey();

    protected abstract String getTaskId();

    protected abstract AtomicInteger getSuccessNum();

    protected abstract AtomicInteger getErrorNum();

    //主逻辑 protected void handleExcel() { //业务筛选校验 do business validate CommonImportValidateRes commonImportValidateRes = doValidateAndInit(); getSuccessNum().addAndGet(commonImportValidateRes.getSuccessNums()); getErrorNum().addAndGet(commonImportValidateRes.getErrorNums());

    //清除业务对象 List sequenceNos = commonImportValidateRes.getSuccessData().stream().map(ImportBean::getSequenceNo).distinct().collect(Collectors.toList()); helper.doClearBusinessData(sequenceNos, getType()); //保存成功数据 helper.saveBusinessData(buildDataMap(commonImportValidateRes.getSuccessData()), getType()); //缓存失败数据 helper.doSaveRedisInfo(getErrorRedisKey(), getTaskId(), commonImportValidateRes.getErrorData()); }

    }

    这里对用户自定义的数据查询，以及导入成功/失败 的数据处理 均做了“抽象”。

    1.4 这里做一个简单的示例：

    @Service("klxcwzBasicImportService")

    @Slf4j

    public class KlxcwzBasicImportService extends AbstractBasicImportListenerService{

    private AtomicInteger totalSuccessNums = new AtomicInteger(0);

    private AtomicInteger totalErrorNums = new AtomicInteger(0);

    private String taskId;

    private UploadInitModel uploadInitModel;

    private List cachedDataList = new ArrayList<>(ExcelConst.MAX_IMPORT_DB_NUMS);

    public KlxcwzBasicImportService(BasicInfoContext context) {

    super(context);

    uploadInitModel = context.getInitModel();

    taskId = context.getTaskId();

    }

    @Override

    public void invoke(KlxcwzModel entity, AnalysisContext context) {

    cachedDataList.add(entity);

    // 触达阈值，业务逻辑操作

    if (cachedDataList.size() >= ExcelConst.MAX_IMPORT_DB_NUMS) {

    handleExcel();

    // 存储完成清理 list

    cachedDataList.clear();

    }

    }

    @Override

    protected AtomicInteger getSuccessNum() {

    return totalSuccessNums;

    }

    @Override

    protected AtomicInteger getErrorNum() {

    return totalErrorNums;

    }

    @Override

    protected String getType() {

    return BasicExcelEnum.KL_XCWZ.getType();

    }

    @Override

    protected String getErrorRedisKey() {

    return BasicExcelEnum.KL_XCWZ.getErrorRedisKey();

    }

    @Override

    protected String getTaskId() {

    return taskId;

    }

    @Override

    public void doAfterAllAnalysed(AnalysisContext analysisContext) {

    //最后一次 不满MAX_IMPORT_DB_NUMS的处理

    handleExcel();

    helper.buildTaskResult(getType(), getTaskId(), getSuccessNum(), getErrorNum());

    }

    protected List> buildDataMap(List successData) {

    List> mapInfos = new ArrayList<>();

    if (CollectionUtils.isEmpty(successData)) {

    return mapInfos;

    }

    List> dataList = new ArrayList<>();

    for (ImportBean bean : successData) {

    Map entityMap = new HashMap<>();

    KlxcwzModel klxcwzModel = (KlxcwzModel) bean;

    helper.doCommonSqlFieldPadding(klxcwzModel, entityMap);

    entityMap.put("id", UUID.randomUUID().toString().replace("-", ""));

    entityMap.put("materialCode",klxcwzModel.getMaterialCode());

    dataList.add(entityMap);

    }

    return dataList;

    }

    protected CommonImportValidateRes doValidateAndInit() {

    List errorData = new ArrayList<>();

    List successData = new ArrayList<>();

    List codes = cachedDataList.stream().map(KlxcwzModel::getMaterialCode).collect(Collectors.toList());

    Map truthData = this.materialCostListService.checkDataFromMaterialCostList(codes);

    for(KlxcwzModel model: cachedDataList){

    StringBuilder sb = new StringBuilder();

    sb.append(model.getMaterialCode());

    String materialCode = model.getMaterialCode();

    MaterialCostModel costModel = MapUtils.getObject(truthData, materialCode);

    boolean errFlag = false;

    if(costModel == null){

    sb.append(" 料号对应的数据不存在;");

    errFlag = true;

    }else{

    if(!model.getMateriaName().equals(costModel.getProductName())){

    sb.append("品名未匹配(参考值:"+costModel.getProductName()+");");

    errFlag = true;

    }

    }

    if(errFlag){

    model.setErrorMsg(sb.toString());

    errorData.add(model);

    continue;

    }

    successData.add(model);

    }

    return helper.buildRes(successData, errorData);

    }

    }

    五.思考

    - 经过上述的方法可以满足用户 对低代码导入功能的 可扩展性的数据校验 - 以上数据校验是需要即时返回错误数据给用户，较为友好的提示出 正确条数/错误条数。 - 如果需要使用低代码产品的校验，这里也可以扩展满足 - 之所以选择此方式，是对后续升级迭代比较友好，另一方面，也是在做扩展时，用户仅需要增加 制定部分即可满足新模型数据的校验。
