自定义实现权限系统

从头实现权限系统

一、需求：

用户不想用系统自带的权限系统，需要从头写一个树形的权限控制系统，这就需要重新设计表单，通过前端二开页面，完成权限系统的设置。

左侧为角色，点击左侧某个角色，右侧权限实时显示角色对应的角色权限。

2、实现方式：

2.1后端设计权限逻辑

2.1.1 角色表

2.1.2 菜单表

应用名称为关联表单，决定该菜单是哪个应用下的数据，菜单编码唯一，上级目录为关联表单，如果该菜单为一级目录，且下面没有二级目录，则上级目录为空，上级目录编码为0，菜单类型为非目录，如果该菜单为一级目录，且下面有二级目录，则菜单类型为目录，则上级目录为空，上级目录编码为0；如果为子菜单，则选择对应的上级目录。

2.1.3 角色权限表

此表关联角色表与菜单表，菜单名称为关联菜单表，映射出选择的菜单编码，如果菜单类型为非目录，则展示权限子表，可以输入按钮权限，按钮权限编码唯一。如果菜单类型为目录，则不显示权限子表。角色名称关联角色表，确定该条权限为哪个角色的权限。

2.2 前端显示

2.2.1 数据获取

通过当前选中角色ID与应用ID 获取菜单表与角色权限表相关数据。

菜单表中存有当前系统所有菜单的数据，角色权限表中存有角色权限的数据。

2.2.2 数据处理

将获取到的角色权限表数据进行处理，处理成antd tree组件中想要的数据，下面为一个可以处理的代码

forPermissions(menu) {

    let returnPermissionList = ([] as any)

    this.originMenuData.forEach(item => {

      if(item.superiorFolderCode ===  menu.menuCode && item.menuTypeCode !== 'Folder') {

        let orgPermission = this.originPermissionsData.filter(ele => {

          return ele.menuCode === item.menuCode

        })[0]

        let permission = {

          name: '',

          id: '',

          key: '',

          isActive: true,

          children: ([] as any)

        };

        if(orgPermission) {

          orgPermission.permission_sheet.forEach(ele => {

            ele.name = ele.permissionsName;

            ele.code = ele.permissionsCode;

            ele.key = ele.code;

            if(!ele.isActive){

              permission.isActive = false

            }

          });

       

          permission.name = orgPermission.menuName.menuNameChinese;

          permission.id = orgPermission.id;

          // permission.isActive = orgPermission.folderIsView;

          permission.children = orgPermission.permission_sheet;

          permission.key = orgPermission.menuCode;

          returnPermissionList.push(permission);

        }

      }

      else if (item.menuTypeCode === 'Folder' && item.superiorFolderCode === menu.menuCode) {

        let orgPermission = this.originPermissionsData.filter(ele => {

          return ele.menuCode === item.menuCode

        })[0];

        if (orgPermission) {

          let permission = {

            name: '',

            key: '',

            isActive: false,

            children: ([] as any)

          };

          permission.name = orgPermission.menuName.menuNameChinese;

          permission.key = orgPermission.menuCode;

          permission.children = this.forPermissions(orgPermission);

          returnPermissionList.push(permission);

        }

      }

    })

    return returnPermissionList;

  }
