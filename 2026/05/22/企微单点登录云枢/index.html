企微单点登录云枢

业务场景：某项目需求是从企微单点登录到云枢（PC,移动端）。

企业配置：只能配置一个携带code的地址，通过code去获取token,登录。
解决思路：

1.设置空白页(不需要token)

注* 只需要设置portal端即可，因为portal 和 移动端的路由差异

2.通过code调用接口拿到token 3.解决一个地址区分跳转PC，移动端

代码：

1.portal：

1) 根据二开规范新增空白页，进行通过code获取token

注*：设置放行这个路由

1.获取地址参数

   getCode(name: any) {

        let reg = new RegExp("(^|&)" + name + "=([^&]*)(&|$)", "i");

        let r = window.location.search.substr(1).match(reg);

        if (r != null) {

            return unescape(r[2]);

        };

        return null;

    }

2.移除token,refreshe_token,expireTime,再通过code获取最新的token等信息

async getBycode(){

    localStorage.removeItem('token')

    localStorage.removeItem('refresh_token')

    localStorage.removeItem('expireTime')

        let code: any = this.getCode('code')

        if(code){

            let parmas = {

                        code: code

                    }

        let res: any = await OAuthApi.getWxcode(parmas)

            if(res.errcode === 0){

               

                    const token:any = (res as any).access_token;

                    const refresh_tokens:any = (res as any).refresh_token;

                    localStorage.setItem('refresh_token', refresh_tokens);

                    localStorage.setItem('token', token);

                   

                    //携带token 做跳转

                    this.getCodeUrl(token);

                 

            }

        }

    }

    async getCodeUrl(token) {

        let theUrl: any = `

        let workflowInstanceId: any = this.getCode('workflowInstanceId')

        let workitemId: any = this.getCode('workitemId')

            if(token){

                if (workflowInstanceId && workitemId) {

                theUrl = ${env.portalHost}/form/detail?;

                    theUrl += workflowInstanceId=${workflowInstanceId}&workitemId=${workitemId || ''};

                }

                window.location.href  = theUrl

            }else{

                this.$router.push({ name: 'login' }).catch((err: any) => {

                console.error(err);

                });

            }

    }
